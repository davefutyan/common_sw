40 T swap_endian(T u)
41 {
42  union
43  {
44  T u;
45  unsigned char u8[sizeof(T)];
46  } source, dest;
47 
48  source.u = u;
49 
50  for (size_t k = 0; k < sizeof(T); k++)
51  dest.u8[k] = source.u8[sizeof(T) - k - 1];
52 
53  return dest.u;
54 };
