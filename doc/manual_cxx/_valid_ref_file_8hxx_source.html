255  ValidRefFile(const std::list<std::string> & fileNames) {
256 
257  std::string structName;
258  FitsDalHeader * header = nullptr;
259  std::pair<std::set<ValidityPeriod>::iterator, bool> returnValue;
260 
261  for (auto & fileName : fileNames) {
262  // First iteration, do not check extname
263  if (header == nullptr) {
264  header = new FitsDalHeader(fileName);
265  }
266  // Subsequent iterations, check that the current reference file's extname
267  // is the same as those of the previous reference files
268  else {
269  header = new FitsDalHeader(fileName);
270 
271  if (structName != header->GetAttr<std::string>("EXTNAME")) {
272  throw std::runtime_error("Found reference files of different types: ["
273  + structName + "], ["
274  + header->GetAttr<std::string>("EXTNAME") + "]");
275  }
276  }
277 
278  structName = header->GetAttr<std::string>("EXTNAME");
279 
280  ValidityPeriod vp(header->GetAttr<std::string>("V_STRT_U"),
281  header->GetAttr<std::string>("V_STOP_U"),
282  header->GetAttr<int32_t>("ARCH_REV"),
283  header->GetAttr<int32_t>("PROC_NUM"),
284  fileName,
285  header->GetFileName());
286 
287  returnValue = m_validityPeriods.insert(vp);
288  // The second evaluates to false if the insertion replaced an existing
289  // item, meaning a validity time with identical archive revision number
290  // and processing number had already been inserted. This is an error.
291  if (!returnValue.second) {

293  errorMsg.append(std::to_string(vp.getArchRev()));
294  errorMsg.append(" and PROC_NUM=");

296  errorMsg.append(" (" + returnValue.first->getFileName() + ", " + fileName + ")");

338  HDU * getFile(const UTC & dataTime) {
339 
340  // Iterate over the reference files' validity periods, starting with the
341  // most recently created file, to find the file containing the specific data time.
342  for (auto validityPeriod : boost::adaptors::reverse(m_validityPeriods)) {
343 
344  if (!validityPeriod.contains(dataTime)) {
345  continue;
346  }
347  // Found the right validity period
348 
349  // First time opening an HDU after this object was created
350  if (m_hdu == nullptr) {
351  m_hdu = new HDU(validityPeriod.getFileName());
352  }
353  // Need to switch to another HDU ?
354  else if (m_hdu->GetFileName() != validityPeriod.getBaseFileName()) {
355  HDU * old_hdu = m_hdu;
356  m_hdu = new HDU(validityPeriod.getFileName());
357  delete old_hdu;
358  }
359 
360  return m_hdu;
361  }
