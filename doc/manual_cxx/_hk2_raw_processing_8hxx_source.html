271  template < class T >
272  void updateParameter(const std::string & colName, const T & value) {
273 
274  // find the table in which the column is present.
275  std::map<std::string, std::map<std::string, ColMetaData *>>::iterator i_colData =
276  m_colData.begin();
277  while (i_colData != m_colData.end()) {
278  std::map<std::string, ColMetaData *>::iterator ii_colData = i_colData->second.find(colName);
279  if (ii_colData != i_colData->second.end()) {
280  // this is the table, we set the value
281  ii_colData->second->setValue(value);
282  return;
283  }
284 
285  i_colData++;
286  }
287 
288  // the colName is unknow. This should not happen, we throw an exception
289  throw std::runtime_error ("Cannot update value for parameter " + colName +
290  ". Unknown parameter.");
291  }
